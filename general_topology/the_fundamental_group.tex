\chapter{The Fundamental Group}
\section{Homotopies}

\begin{proposition}
	Being homotopic is a congruence on $\cat{Top}$.
\end{proposition}

\begin{proof}
	First we show that being homotopic induces an equivalence relation on 
	\begin{equation*}
		A := \bigcup_{(X,Y) \in \ob(\cat{Top})\times \ob(\cat{Top})} C(X,Y).
	\end{equation*}
\end{proof}
