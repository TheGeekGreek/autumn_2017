\chapter{Constructions}
\section{Limits}



\begin{definition}[Product]
	Let $\Ccal$ be a category and $A$ a set. Define $\cat{A}$ to be the discrete category with $\ob(\cat{A}) := A$. Moreover, let $D$ be a diagram in $\Ccal$ of shape $\cat{A}$. A \bld{(small) product of $D$} is a limit of $D$. 
	\label{def:product}
\end{definition}

\begin{remark}
	In the setting of definition \ref{def:product}, $D$ yields a family $(X_\alpha)_{\alpha \in A}$ in $\Ccal$ and thus we we speak also of a product of the family $(X_\alpha)_{\alpha \in A}$. If a product exists in $\Ccal$, we write
	\begin{equation}
		\prod_{\alpha \in A} X_\alpha := \lim_{\leftarrow \cat{A}} D.
	\end{equation}
\end{remark}
