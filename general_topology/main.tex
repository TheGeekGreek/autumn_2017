\chapter{Set Theory}
\section*{Basic Concepts}

\begin{problem}
	Let $n \in \mathbb{N}$ and $a_{kj} \in \mathbb{C}$ for $k = 0,\dots,n + 1$, $j = 0,\dots,n$. Show that
	\begin{equation*}
		\sum_{k = 0}^{n+1}\sum_{j = 0}^n a_{kj} = \sum_{0 \leq k \leq j \leq n}a_{kj} + \sum_{0 \leq j < k \leq n+1} a_{kj}.
	\end{equation*}
\end{problem}
