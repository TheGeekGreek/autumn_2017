\begin{definition}[Representable and Corepresentable Functor]
	Let $\Ccal$ be a locally small category. A covariant functor $F$ is said to be \bld{representable}, if there exists an object $X \in \Ccal$, such that $F \cong \Hom_\Ccal(X,-)$. A contravariant functor $F$ is said to be \bld{corepresentable}, if there exists $X \in \Ccal$ such that $F \cong \Hom_\Ccal(-,X)$, where $\Hom_\Ccal(-,X) := \Hom_{\Ccal^\oprm}(X,-)$. 
\end{definition}

The functor defined in the dualized statement of proposition \ref{prop:H_bullet} has its own name.

\begin{definition}[Yoneda embedding of $\bm{\Ccal}$]
	Let $\Ccal$ be a locally small category. Then the functor $H_\bullet := H^\bullet : \Ccal \to \sbr{\Ccal^\oprm,\cat{Set}}$ is called the \bld{Yoneda embedding of $\Ccal$}.  
\end{definition}

\begin{theorem}[Yoneda Lemma]
	Let $\Ccal$ be a locally small category. For any functor $F : \Ccal \to \cat{Set}$ and for every object $X \in \Ccal$ there is a bijection
	\begin{equation}
		\sbr{\Ccal,\cat{Set}}\del[1]{\Hom_\Ccal(X,-),F} \cong F(X)	
	\end{equation}
	\noindent that associates to each $\alpha : \Hom_\Ccal(X,-) \Rightarrow F$ the element $\alpha_X(\id_X) \in F(X)$. Moreover, the correspondence is natural in both $X$ and $F$.
	\label{thm:Yoneda}
\end{theorem}
